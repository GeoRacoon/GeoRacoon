12def update_view(data: NDArray, view: tuple[int, int, int, int], block: ArrayLike) -> None: 

13 """Update a view from the data array with a block 

14 

15 The block must exactly match the shape of the view: 

16 `block.shape == (view[3], view[2])`, where 

17 `view = (x, y, width, height)`. 

18 

19 Parameters 

20 ---------- 

21 data : NDArray 

22 The array that we want to update 

23 view : tuple[int, int, int, int] 

24 tuple (x, y, width, height) defining the view of the data array to update 

25 block : ArrayLike 

26 np.array with the updated values. 

27 

28 Returns 

29 -------- 

30 None 

31 

32 See Also 

33 -------- 

34 :func:`~riogrande.prepare.get_view` : Read a rectangular view from an array. 

35 :func:`~riogrande.prepare.create_views` : Generate a set of views covering an array. 

36 

37 Examples 

38 -------- 

39 >>> data = np.zeros((5, 5)) 

40 >>> block = np.ones((2, 3)) 

41 >>> update_view(data, (1, 2, 3, 2), block) 

42 >>> data 

43 array([[0., 0., 0., 0., 0.], 

44 [0., 0., 0., 0., 0.], 

45 [0., 1., 1., 1., 0.], 

46 [0., 1., 1., 1., 0.], 

47 [0., 0., 0., 0., 0.]]) 

48 """ 

49 data[slice(view[1], view[1] + view[3]), slice(view[0], view[0] + view[2])] = block

52def create_views(view_size: tuple[int, int], border: tuple[int, int], size: tuple[int, int]) -> tuple[list, ...]: 

53 """Returns a set of views on which an operation can be applied independently 

54 

55 Parameters 

56 ---------- 

57 view_size : tuple[int, int] 

58 The size (width, height) in pixels of a single view (excluding borders) 

59 border : tuple[int, int] 

60 The border size (width, height) in number of pixels along each axis 

61 size : tuple[int, int] 

62 The total size of the map in number of pixels (width, height) 

63 

64 Return 

65 ------ 

66 tuple 

67 The first element is a list of tuples (x, y, width, height) defining each 

68 view on which to apply an operation. 

69 The second element is a list of tuples (x, y, width, height) defining for 

70 each view the usable region. 

71 A region is usable if it does not contain any artificial border effects 

72 that were introduced from splitting up a bigger view into smaller chunks 

73 

74 Notes 

75 ----- 

76 - Handles cases where the region cannot be divided evenly by `view_size`. 

77 The last row/column of views may be smaller (`leftovers`) and are still included. 

78 Uses :func:`math.floor` to compute the number of full blocks. 

79 - Borders on the outer edges are reduced to fit within the total size. 

80 

81 See Also 

82 -------- 

83 :func:`~riogrande.prepare.update_view` : Write a block into a view of an array. 

84 :func:`~riogrande.prepare.get_view` : Read a rectangular view from an array. 

85 :func:`~riogrande.prepare.relative_view` : Express an inner view relative to an outer view. 

86 

87 Examples 

88 -------- 

89 >>> views, usable = create_views((5, 5), (1, 1), (9, 9)) 

90 >>> len(views), len(usable) 

91 (4, 4) 

92 >>> views 

93 [(0, 0, 6, 6), (3, 0, 6, 5), (0, 3, 6, 6), (3, 3, 6, 6)] 

94 >>> usable 

95 [(0, 0, 4, 4), (4, 0, 5, 4), (0, 4, 4, 5), (4, 4, 5, 5)] 

96 """ 

97 assert all(len(x) == 2 for x in (view_size, border, size)), \ 

98 f"{len(view_size)=},{len(border)=},{len(size)=} all need to be of " \ 

99 "length 2 (width, height)" 

100 # calculate the leftovers along both axes 

101 leftovers = list(map(lambda x: x[0] % x[1], zip(size, view_size))) 

102 # number of full block along each axis that do not cover the leftovers 

103 nbr_views = list(map(lambda x: math.floor((x[1] - x[2]) / x[0]), 

104 zip(view_size, size, leftovers))) 

105 

106 xstarts = [] 

107 ystarts = [] 

108 heights = [] 

109 widths = [] 

110 inner_xs = [] 

111 inner_ys = [] 

112 inner_h = [] 

113 inner_w = [] 

114 for i in range(nbr_views[0]): # horizontally 

115 if 0 < i < nbr_views[0] - 1: 

116 hpadding = 2 * border[0] 

117 elif nbr_views[0] == 1: 

118 hpadding = 0 

119 else: 

120 hpadding = border[0] 

121 for j in range(nbr_views[1]): # vertically 

122 if 0 < j < nbr_views[1] - 1: 

123 vpadding = 2 * border[1] 

124 elif nbr_views[1] == 1: 

125 vpadding = 0 

126 else: 

127 vpadding = border[1] 

128 xstarts.append(max(0, i * view_size[0] - border[0])) 

129 widths.append(view_size[0] + hpadding) 

130 ystarts.append(max(0, j * view_size[1] - border[1])) 

131 heights.append(view_size[1] + vpadding) 

132 # the useable inner view 

133 # starting points form a regular grid 

134 inner_xs.append(i * view_size[0]) 

135 inner_ys.append(j * view_size[1]) 

136 # used is always the view_size 

137 inner_w.append(view_size[0]) 

138 inner_h.append(view_size[1]) 

139 # handle the leftover 

140 # prepare for horiz. leftover pixels 

141 if i == nbr_views[0] - 1 and leftovers[0]: 

142 inner_w[-1] = leftovers[0] 

143 widths[-1] = leftovers[0] + 2 * border[0] 

144 # prepare for vertical. leftover pixels 

145 if j == nbr_views[1] - 1 and leftovers[1]: 

146 inner_h[-1] = leftovers[1] 

147 heights[-1] = leftovers[1] + 2 * border[1] 

148 if leftovers[0]: 

149 # add the last column of leftovers 

150 for j in range(nbr_views[1]): # vertically since column 

151 _width = border[0] 

152 if 0 < j < nbr_views[1] - 1: 

153 vpadding = 2 * border[1] 

154 else: 

155 vpadding = border[1] 

156 xstarts.append(size[0] - (view_size[0] + _width)) 

157 ystarts.append(max(0, j * view_size[1] - border[1])) 

158 widths.append(view_size[0] + _width) 

159 heights.append(view_size[1] + vpadding) 

160 # the useable inner view 

161 inner_xs.append(size[0] - view_size[0]) 

162 inner_ys.append(max(0, j * view_size[1])) 

163 inner_w.append(view_size[0]) 

164 inner_h.append(view_size[1]) 

165 # NOTE: this is not the corner, but the last of the regular blocks 

166 if j == nbr_views[1] - 1 and leftovers[1]: 

167 inner_h[-1] = leftovers[1] 

168 heights[-1] = leftovers[1] + border[1] 

169 if leftovers[1]: 

170 # add the last row of leftovers 

171 for i in range(nbr_views[0]): # horizontally since row 

172 _height = border[1] 

173 if 0 < i < nbr_views[0] - 1: 

174 hpadding = 2 * border[0] 

175 else: 

176 hpadding = border[0] 

177 xstarts.append(max(0, i * view_size[0] - border[0])) 

178 ystarts.append(size[1] - (view_size[1] + _height)) 

179 widths.append(view_size[0] + hpadding) 

180 heights.append(view_size[1] + _height) 

181 # the useable inner view 

182 inner_xs.append(i * view_size[0]) 

183 inner_ys.append(size[1] - view_size[1]) 

184 inner_w.append(view_size[0]) 

185 inner_h.append(view_size[1]) 

186 if i == nbr_views[0] - 1 and leftovers[0]: 

187 inner_w[-1] = leftovers[0] 

188 widths[-1] = leftovers[0] + 2 * border[0] 

189 if all(leftovers): 

190 # add the outer corner block 

191 xstarts.append(size[0] - (view_size[0] + border[0])) 

192 ystarts.append(size[1] - (view_size[1] + border[1])) 

193 widths.append(view_size[0] + border[0]) 

194 heights.append(view_size[1] + border[1]) 

195 inner_xs.append(size[0] - view_size[0]) 

196 inner_ys.append(size[1] - view_size[1]) 

197 inner_w.append(view_size[0]) 

198 inner_h.append(view_size[1]) 

199 

200 return (list(zip(xstarts, ystarts, widths, heights)), 

201 list(zip(inner_xs, inner_ys, inner_w, inner_h)))

204def get_view(data: NDArray, view: tuple[int, int, int, int]) -> NDArray: 

205 """Return a recatangular view of the data array 

206 

207 .. note:: 

208 data.shape == height, width! 

209 

210 Parameters 

211 ---------- 

212 data : NDArray 

213 np.array to return the view from 

214 view : tuple[int, int, int, int] 

215 tuple (x, y, width, height) defining the view 

216 

217 Returns 

218 ------- 

219 NDArray 

220 A view (slice) of `data` with shape `(height, width)` as specified 

221 by the `view` tuple. 

222 

223 See Also 

224 -------- 

225 :func:`~riogrande.prepare.update_view` : Write a block into a view of an array. 

226 :func:`~riogrande.prepare.create_views` : Generate a set of views covering an array. 

227 

228 Examples 

229 -------- 

230 >>> arr = np.arange(16).reshape(4, 4) 

231 >>> arr 

232 array([[ 0, 1, 2, 3], 

233 [ 4, 5, 6, 7], 

234 [ 8, 9, 10, 11], 

235 [12, 13, 14, 15]]) 

236 >>> get_view(arr, (1, 1, 2, 2)) 

237 array([[ 5, 6], 

238 [ 9, 10]]) 

239 """ 

240 return data[slice(view[1], view[1] + view[3]), 

241 slice(view[0], view[0] + view[2])]

244def relative_view(view: tuple[int, int, int, int], 

245 inner_view: tuple[int, int, int, int]) -> tuple[int, int, int, int]: 

246 """Return the `inner_view` relative to `view` 

247 

248 Given two rectangular regions defined as `(x, y, width, height)`, 

249 this function returns the coordinates of `inner_view` relative 

250 to the origin of `view`. 

251 

252 Parameters 

253 ---------- 

254 view : tuple[int, int, int, int] 

255 A 4-tuple `(x, y, width, height)` defining the outer view. 

256 inner_view : tuple[int, int, int, int] 

257 A 4-tuple `(x, y, width, height)` defining a subregion inside `view`. 

258 

259 Returns 

260 -------- 

261 tuple 

262 A 4-tuple `(x, y, width, height)` giving the position and size of 

263 `inner_view` relative to `view`. The width and height are unchanged. 

264 

265 See Also 

266 -------- 

267 :func:`~riogrande.prepare.create_views` : Generate outer and inner view pairs. 

268 

269 Examples 

270 -------- 

271 >>> outer = (10, 20, 100, 50) 

272 >>> inner = (15, 30, 20, 10) 

273 >>> outer 

274 (10, 20, 100, 50) 

275 >>> inner 

276 (15, 30, 20, 10) 

277 >>> relative_view(outer, inner) 

278 (5, 10, 20, 10) 

279 """ 

280 return (inner_view[0] - view[0], inner_view[1] - view[1], 

281 inner_view[2], inner_view[3])
